237  // All audio threads should be stopped now.

396 APF_MIMOPROCESSOR_BASE::MimoProcessor(const parameter_map& params_)
397  : interface_policy(params_)
398  , query_policy(params_.get("fifo_size", 1024))
399  , params(params_)
400  , _fifo(params.get("fifo_size", 1024))
401  , _current_list(nullptr)
402  , _num_threads(params.get("threads"
403  , thread_policy::default_number_of_threads()))
404  , _input_list(_fifo)
405  , _output_list(_fifo)
406 {
407  assert(_num_threads > 0);
408 
409  // deactivate FIFO for non-realtime initializations
410  if (!_fifo.deactivate()) throw std::logic_error("Bug: FIFO not empty!");
411 
412  // Create N-1 worker threads. NOTE: Number 0 is reserved for the main thread.
413  _thread_data.reserve(_num_threads - 1);
414  for (int i = 1; i < _num_threads; ++i)
415  {
416  _thread_data.emplace_back(i, *this);
417  }
418 }

468 APF_MIMOPROCESSOR_BASE::_process_current_list_in_main_thread()
469 {
470  assert(_current_list);
471  if (_current_list->empty()) return;
472 
473  // wake all threads
474  for (auto& it: _thread_data) it.cont_semaphore.post();
475 
476  _process_selected_items_in_current_list(0);

478  // wait for worker threads
