115template<typename T>
116void
117pointer_policy<T*>::audio_callback(size_t n, T* const* in, T* const* out)
118{
119 assert(n == this->block_size());
120 (void)n; // avoid "unused parameter" warning
121
122 _in = in;
123 _out = out;
124 this->process();
125}
