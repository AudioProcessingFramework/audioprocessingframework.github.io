220  bool connect_pending_connections() const
221  {
222  _pending_connections_t still_pending_connections;
223 
224  APF_JACKCLIENT_DEBUG_MSG("Connecting " << _pending_connections.size()
225  << " pending connections ...");
226  while (_pending_connections.size() > 0)
227  {
228  _connect_ports_helper(_pending_connections.back().first
229  , _pending_connections.back().second, still_pending_connections);
230  _pending_connections.pop_back();
231  }
232  APF_JACKCLIENT_DEBUG_MSG("Still pending connections: "
233  << still_pending_connections.size());
234 
235  _pending_connections.swap(still_pending_connections);
236  return _pending_connections.empty();
237  }

420  bool _connect_ports_helper(const std::string& source
421  , const std::string& destination
422  , _pending_connections_t& pending_connections) const
423  {
424  if (_client == nullptr) return false;
425  int success = jack_connect(_client, source.c_str(), destination.c_str());
426  switch (success)
427  {
428  case 0:
429  break;
430  case EEXIST:
431  APF_JACKCLIENT_DEBUG_MSG("Connection already exists! ("
432  << source << " -> " << destination << ")");
433  break;
434  default:
435  APF_JACKCLIENT_DEBUG_MSG("Unable to connect "
436  << source << " -> " << destination
437  << "! Adding this to pending connections ...");
438  pending_connections.push_back(std::make_pair(source, destination));
439  // TODO: return something else than true/false?
440  return false;
441  }
442  return true;
443  }

464 JackClient::JackClient(const std::string& name
465  , callback_usage_t callback_usage)
466  : _client_name(name)
467  , _client(nullptr) // will be set after connecting to JACK
468  , _sample_rate(0) // -- " --
469  , _buffer_size(0) // -- " --
470 {
471  // check if client name is too long
472  // jack_client_name_size() returns the size including the terminating \0
473  // character, hence the ">=".

476  {
477  throw jack_error("Client name is too long! ('" + name + "')");
478  }
479 
480  // TODO: make parameters for these options:
481  //jack_options_t options = JackNoStartServer;
482  jack_options_t options = JackUseExactName;
483  //jack_options_t options = JackNullOption;
484 
485  // JackNoStartServer: see also JACK_NO_START_SERVER
486  // TODO: JackServerName? needs extra parameter in jack_client_open()!
487 
488  jack_status_t status;
489  _client = jack_client_open(name.c_str(), options, &status);
490  if (!_client) throw jack_error(status);
491 
492  if (options & JackUseExactName)
493  {
494  assert(_client_name == jack_get_client_name(_client));
495  }
496  else
497  {
498  _client_name = jack_get_client_name(_client);
499  }
500 
501  // TODO: error callback
502  //jack_set_error_function(default_jack_error_callback);
503 
504  if (callback_usage == use_jack_process_callback)
505  {

507  {
508  throw jack_error("Could not set process callback function for '"
